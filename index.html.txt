            loadNotes() {
                if (!this.currentUser) return;
                
                const savedNotes = localStorage.getItem(this.userStorageKey);
                
                if (savedNotes) {
                    try {
                        this.notes = JSON.parse(savedNotes);
                    } catch (e) {
                        console.error('Âä†ËΩΩÁ¨îËÆ∞Â§±Ë¥•:', e);
                        this.notes = [];
                    }
                }
            }

            }
            
            // ÂØºÂá∫Á¨îËÆ∞‰∏∫JSONÊñá‰ª∂
            exportNotes() {
                if (this.notes.length === 0) {
                    alert('Ê≤°ÊúâÂèØÂØºÂá∫ÁöÑÁ¨îËÆ∞');
                    return;
                }
                
                const dataStr = JSON.stringify(this.notes, null, 2);
                const dataBlob = new Blob([dataStr], {type: 'application/json'});
                
                const url = URL.createObjectURL(dataBlob);
                const link = document.createElement('a');
                link.href = url;
                link.download = `lano-plus-notes-${new Date().toISOString().split('T')[0]}.json`;
                document.body.appendChild(link);
                link.click();
                document.body.removeChild(link);
                URL.revokeObjectURL(url);
                
                alert('Á¨îËÆ∞ÂØºÂá∫ÊàêÂäüÔºÅ');
            }

                reader.onload = (e) => {
                    try {

                        if (!Array.isArray(importedNotes)) {
                            throw new Error('Êñá‰ª∂Ê†ºÂºè‰∏çÊ≠£Á°Æ');
                        }
                        
                        if (confirm(`Á°ÆÂÆöË¶ÅÂØºÂÖ• ${importedNotes.length} Êù°Á¨îËÆ∞ÂêóÔºüËøôÂ∞ÜË¶ÜÁõñÊÇ®ÂΩìÂâçÁöÑÁ¨îËÆ∞„ÄÇ`)) {
                            this.notes = importedNotes;
                            this.saveNotes();
                            this.updateNotesList();
                            alert('Á¨îËÆ∞ÂØºÂÖ•ÊàêÂäüÔºÅ');
                        }
                    } catch (error) {
                        alert('ÂØºÂÖ•Â§±Ë¥•ÔºöÊñá‰ª∂Ê†ºÂºè‰∏çÊ≠£Á°Æ');
                        console.error('ÂØºÂÖ•ÈîôËØØ:', error);

                };
                
                reader.readAsText(file);
            }
